var slideH;
// 함수--------------------------------------------------------
var slideUpFn = function(height){
  var setH = height;
  slideH = setInterval(function(){
      setH -= 1;
      jsContentArea.style.height = setH + 'px';
      if(setH >= 0) {
        jsContentArea.style.height = setH + 'px'
      } else{
        clearInterval(slideH) ;
        // jsContentArea.style = null;
        jsContentArea.style.display = 'none';
      }
  },timed/100);
};
